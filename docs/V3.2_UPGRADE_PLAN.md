## 二、P0 核心修复方案

### 3.2.1 BM25 中文分词替换（jieba）

#### 现状诊断

当前 BM25 分词器（`retrieval_service.py` 第 155-160 行）使用正则 `[\u4e00-\u9fff]` 逐字拆分中文：

```python
# 当前实现 —— 每个汉字被拆成独立 token
def _tokenize(text: str) -> List[str]:
    tokens = re.findall(r'[\u4e00-\u9fff]|[a-zA-Z]+', text.lower())
    return tokens if tokens else text.lower().split()
```

**致命后果**：以"机器学习"为例，被切成 `["机","器","学","习"]`，BM25 的 IDF 权重完全失效。"学习"和"机器"在 BM25 眼里永远是四个单字，无法区分"机器学习"和"学习机器"。中文专业术语（如"反向传播""注意力机制""向量检索"）的召回率几乎为零。

#### 最优解

替换为 **jieba 分词**——中文 NLP 事实标准，支持精确模式、自定义词典、词性标注。

```python
# 升级后实现
import jieba

def _tokenize(text: str) -> List[str]:
    """中文 jieba 分词 + 英文单词提取"""
    tokens = []
    # jieba 精确模式分词
    for word in jieba.cut(text):
        word = word.strip().lower()
        if word and not word.isspace():
            tokens.append(word)
    return tokens
```

### 3.2.2 语义感知分块（Semantic Chunking）

#### 现状诊断

当前 PyPDF 路径使用 `RecursiveCharacterTextSplitter(chunk_size=512, overlap=100)` 固定分块：

```python
# mineru_document_service.py:179-184
splitter = RecursiveCharacterTextSplitter(
    chunk_size=settings.CHUNK_SIZE,        # 512
    chunk_overlap=settings.CHUNK_OVERLAP,  # 100
    separators=["\n\n", "\n", "。", "！", "？", "；", " ", ""],
)
```

**核心问题**：512 字符是盲切——完全不考虑文档的自然语义边界。一段完整的论述可能被切到两个 chunk，检索时只能拿到半句话。

#### 最优解

**基于相邻句子 Embedding 余弦相似度的自适应分块**——当相邻句子的语义相似度骤降时（即话题发生切换），在该处切分。

```python
# 新增文件: backend/app/services/semantic_splitter.py

from typing import List
import numpy as np
from sentence_transformers import SentenceTransformer


class SemanticTextSplitter:
    """基于相邻句子 Embedding 相似度的自适应分块器。
    
    核心思想：
    1. 将文本按句子切分
    2. 计算相邻句子的 Embedding 余弦相似度
    3. 在相似度骤降处（话题切换点）切分
    4. 同时保证 chunk 不超过最大长度限制
    
    面试要点：能解释为什么语义分块优于固定长度分块——
    固定长度可能在句子中间截断，语义分块在话题边界切分，保证每个 chunk 是完整语义单元。
    """
    
    def __init__(
        self,
        model_name: str = "all-MiniLM-L6-v2",
        similarity_threshold: float = 0.5,
        max_chunk_size: int = 1024,
        min_chunk_size: int = 100,
        overlap_sentences: int = 1,
    ):
        self.model = SentenceTransformer(model_name)
        self.similarity_threshold = similarity_threshold
        self.max_chunk_size = max_chunk_size
        self.min_chunk_size = min_chunk_size
        self.overlap_sentences = overlap_sentences

    def split(self, text: str) -> List[str]:
        """将文本按语义边界切分为 chunks。"""
        # 第 1 步：按句子切分
        sentences = self._split_sentences(text)
        if len(sentences) <= 1:
            return [text] if text.strip() else []

        # 第 2 步：计算相邻句子 Embedding 相似度
        embeddings = self.model.encode(sentences)
        similarities = self._compute_adjacent_similarities(embeddings)

        # 第 3 步：在相似度骤降处标记断点
        breakpoints = self._find_breakpoints(sentences, similarities)

        # 第 4 步：按断点合并句子成 chunks
        chunks = self._merge_sentences(sentences, breakpoints)

        return chunks

    def _split_sentences(self, text: str) -> List[str]:
        """中文句子切分。"""
        import re
        # 按中文标点 + 换行切分
        raw = re.split(r'(?<=[。！？；\n])\s*', text)
        return [s.strip() for s in raw if s.strip()]

    def _compute_adjacent_similarities(self, embeddings: np.ndarray) -> np.ndarray:
        """计算相邻句子 Embedding 的余弦相似度。"""
        n = len(embeddings)
        similarities = np.zeros(n - 1)
        for i in range(n - 1):
            a, b = embeddings[i], embeddings[i + 1]
            sim = np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b) + 1e-8)
            similarities[i] = sim
        return similarities

    def _find_breakpoints(
        self, sentences: List[str], similarities: np.ndarray
    ) -> List[int]:
        """在相似度骤降处标记断点。
        
        断点条件：
        1. 相邻句子相似度 < threshold（话题切换）
        2. 当前 chunk 累计长度 > max_chunk_size（防止过大）
        """
        breakpoints = []
        current_length = 0

        for i, sim in enumerate(similarities):
            current_length += len(sentences[i])
            # 条件 1: 语义相似度骤降
            if sim < self.similarity_threshold and current_length >= self.min_chunk_size:
                breakpoints.append(i + 1)
                current_length = 0
            # 条件 2: 累计长度超过上限，强制切分
            elif current_length >= self.max_chunk_size:
                breakpoints.append(i + 1)
                current_length = 0

        return breakpoints

    def _merge_sentences(
        self, sentences: List[str], breakpoints: List[int]
    ) -> List[str]:
        """按断点合并句子，保留 overlap 句子做上下文衔接。"""
        chunks = []
        start = 0

        for bp in breakpoints:
            chunk_text = "".join(sentences[start:bp])
            if chunk_text.strip():
                chunks.append(chunk_text)
            # overlap：下一个 chunk 从前一个 chunk 的末尾句子开始
            start = max(bp - self.overlap_sentences, start)

        # 最后一个 chunk
        last_chunk = "".join(sentences[start:])
        if last_chunk.strip():
            chunks.append(last_chunk)

        return chunks
```

### 3.2.3 MinerU 路径最大块限制

#### 现状诊断

`_markdown_sections()` 按 H1/H2/H3 标题切分后，没有任何最大块大小检查。一个 H2 章节可能包含 5000+ 字符的连续文本，远超 Embedding 模型的最佳窗口。

#### 修复方案

```python
# mineru_document_service.py: _split_markdown() 方法修改

def _split_markdown(self, markdown: str, document_name: str) -> List[DocumentChunk]:
    """基于标题层级分割，超大 section 做二次递归切分。"""
    sections = self._markdown_sections(markdown)
    chunks: List[DocumentChunk] = []
    chunk_idx = 0

    for section in sections:
        content = section["content"].strip()
        if len(content) < settings.MIN_CHUNK_CHARS:
            continue

        # ── v3.2 升级：超大 section 二次切分 ──
        if len(content) > settings.MAX_CHUNK_CHARS:
            sub_chunks = self._split_large_section(
                content, 
                section.get("h1", ""), 
                section.get("h2", ""), 
                section.get("h3", ""),
                document_name,
                section.get("page", 0),
            )
            for sub in sub_chunks:
                sub.chunk_index = chunk_idx
                chunks.append(sub)
                chunk_idx += 1
        else:
            chunks.append(DocumentChunk(
                chunk_id=uuid4().hex,
                content=content,
                document_name=document_name,
                page=section.get("page", 0),
                chunk_index=chunk_idx,
                metadata={
                    "parser": "mineru",
                    "h1": section.get("h1", ""),
                    "h2": section.get("h2", ""),
                    "h3": section.get("h3", ""),
                },
            ))
            chunk_idx += 1

    return chunks

def _split_large_section(
    self, content: str, h1: str, h2: str, h3: str, 
    document_name: str, page: int
) -> List[DocumentChunk]:
    """对超大 section 做二次递归切分，保留层级元数据。"""
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=settings.MAX_CHUNK_CHARS,
        chunk_overlap=settings.CHUNK_OVERLAP,
        separators=["\n\n", "\n", "。", "！", "？", "；", " ", ""],
    )
    from langchain.schema import Document
    sub_docs = splitter.split_documents([Document(
        page_content=content,
        metadata={"page": page}
    )])
    
    chunks = []
    for sub in sub_docs:
        sub_content = sub.page_content.strip()
        if len(sub_content) < settings.MIN_CHUNK_CHARS:
            continue
        chunks.append(DocumentChunk(
            chunk_id=uuid4().hex,
            content=sub_content,
            document_name=document_name,
            page=page,
            chunk_index=0,  # 由调用方设置
            metadata={
                "parser": "mineru",
                "h1": h1,
                "h2": h2,
                "h3": h3,
                "is_sub_chunk": True,
            },
        ))
    return chunks
```

### 3.2.4 上下文首尾相关排序（Anti-Lost-in-the-Middle）

#### 现状诊断

`_build_messages()` 按 CRAG 分数降序拼接所有 chunk，最相关的 chunk 排在中间位置。GPT-4/DeepSeek 的研究已证实 "Lost in the Middle" 效应——LLM 对中间位置的内容关注度最低。

#### 修复方案

```python
# rag_orchestrator.py: _build_messages() 方法修改

def _build_messages(
    self,
    query: str,
    docs: list[dict],
    history: Sequence[dict] | None,
) -> list[dict]:
    if not docs:
        context_text = "（未检索到可靠参考文档）"
    else:
        # ── v3.2 升级：首尾相关排序（对抗 Lost in the Middle） ──
        # 策略：最相关→开头，次相关→结尾，其他→中间降序
        sorted_docs = sorted(docs, key=lambda d: float(d.get("crag_score", d.get("score", 0))), reverse=True)
        if len(sorted_docs) >= 3:
            reordered = [sorted_docs[0]]  # 最相关 → 开头
            middle = sorted_docs[1:-1]     # 中等 → 中间
            reordered.extend(middle)
            reordered.append(sorted_docs[-1])  # 最后相关 → 结尾
        else:
            reordered = sorted_docs

        context_parts = []
        for index, doc in enumerate(reordered):
            context_parts.append(
                f"[{index + 1}] 来源: {doc.get('document_name', '未知')}, "
                f"第{doc.get('page', '?')}页\n{doc.get('content', '')}"
            )
        context_text = "\n\n---\n\n".join(context_parts)

    messages = list(history or [])[-settings.RAG_MAX_HISTORY_MESSAGES:]
    messages.append({
        "role": "user",
        "content": (
            f"【参考文档】\n{context_text}\n\n"
            f"【用户问题】\n{query}\n\n"
            f"请基于参考文档回答。注意：参考文档按重要性排序，"
            f"请重点关注开头和结尾的文档。"
        ),
    })
    return messages
```

### 3.2.5 CRAG 动态重写


```python
# crag_service.py: process() 方法修改

async def process(self, query: str, docs: list[dict]) -> CRAGResult:
    if not settings.CRAG_ENABLED or not docs:
        return CRAGResult(docs=docs, should_retry=False)

    evaluated = await self._evaluate(query, docs[: settings.RERANKER_TOP_K])
    incorrect_count = sum(
        1 for item in evaluated
        if float(item.get("crag_score", 0.0)) <= settings.CRAG_INCORRECT_THRESHOLD
    )
    incorrect_ratio = incorrect_count / max(len(evaluated), 1)
    should_retry = incorrect_ratio >= settings.CRAG_RETRY_INCORRECT_RATIO

    if should_retry:
        # ── v3.2 升级：动态生成针对性重写 query ──
        # 让 CRAG 评估 LLM 在输出评分的同时，输出一个针对性重写 query
        rewrite_query = await self._generate_rewrite_query(query, evaluated)
        if not rewrite_query:
            # 降级：使用通用重写（比硬编码后缀好）
            rewrite_query = self._fallback_rewrite(query, evaluated)
        return CRAGResult(
            docs=evaluated,
            should_retry=True,
            rewrite_query=rewrite_query,
        )

    filtered = [
        item for item in evaluated
        if float(item.get("crag_score", 0.0)) > settings.CRAG_INCORRECT_THRESHOLD
    ]
    return CRAGResult(docs=filtered or evaluated, should_retry=False)

async def _generate_rewrite_query(
    self, query: str, evaluated: list[dict]
) -> str | None:
    """让 LLM 根据评估结果动态生成针对性重写 query。
    
    核心思想：不是硬编码后缀，而是让 LLM 分析：
    - 哪些文档被判定为 incorrect？为什么？
    - 原始 query 缺少什么关键信息？
    - 应该怎么改写才能命中正确的文档？
    """
    incorrect_docs = [
        d for d in evaluated
        if float(d.get("crag_score", 0.0)) <= settings.CRAG_INCORRECT_THRESHOLD
    ]
    incorrect_summary = "\n".join(
        f"- 文档 {d.get('chunk_id', '?')[:8]}: 内容片段 {str(d.get('content', ''))[:100]}"
        for d in incorrect_docs[:3]
    )

    prompt = (
        f"以下检索结果与用户问题不相关，请分析原因并生成一个更好的检索查询。"
        f"只输出 JSON: {{\"analysis\": \"简要分析\", \"rewrite_query\": \"改写后的查询\"}}\n\n"
        f"用户问题: {query}\n"
        f"不相关文档: {incorrect_summary}\n\n"
        f"要求: 改写后的查询应该更具体、包含关键词、去除模糊表述。"
    )

    try:
        raw = await llm_service.chat(
            messages=[{"role": "user", "content": prompt}],
            system_prompt="你是检索查询优化器，只输出 JSON。",
        )
        import re, json
        match = re.search(r"\{[\s\S]*\}", raw.strip())
        if match:
            data = json.loads(match.group(0))
            return data.get("rewrite_query", "")
    except Exception as exc:
        logger.warning("CRAG 动态重写失败: %s", exc)

    return None

def _fallback_rewrite(self, query: str, evaluated: list[dict]) -> str:
    """降级方案：基于评估结果做规则重写。"""
    incorrect_reasons = set()
    for d in evaluated:
        reason = d.get("crag_reason", "")
        if reason and float(d.get("crag_score", 0.0)) <= settings.CRAG_INCORRECT_THRESHOLD:
            incorrect_reasons.add(reason)

    # 规则驱动的重写增强
    enhancements = []
    if any("定义" in r or "概念" in r for r in incorrect_reasons):
        enhancements.append("定义")
    if any("步骤" in r or "流程" in r for r in incorrect_reasons):
        enhancements.append("步骤流程")
    if any("条件" in r or "前提" in r for r in incorrect_reasons):
        enhancements.append("前提条件")
    if any("示例" in r or "举例" in r for r in incorrect_reasons):
        enhancements.append("示例")

    if enhancements:
        return f"{query} {' '.join(enhancements)}"
    return f"{query} 详细说明"
```

### 3.2.7 Chunk 邻居扩展

#### 现状诊断

检索到 Top-K 后，没有把相邻 chunk 也拉进来。在长文档场景下，答案常分布在连续的几个 chunk 中。

#### 修复方案

```python
# rag_orchestrator.py: 新增方法

async def _expand_with_neighbors(
    self, docs: list[dict], expand: int = 1
) -> list[dict]:
    """为每个检索到的 chunk 拉取相邻 chunk。
    
    Args:
        docs: 检索结果列表，每项需包含 document_name 和 chunk_index
        expand: 向两侧各扩展几个 chunk（默认 1 = 前后各 1 个）
    
    Returns:
        扩展后的结果列表（去重）
    """
    if not docs or expand <= 0:
        return docs

    seen_ids = {d.get("chunk_id", "") for d in docs}
    expanded = list(docs)

    for doc in docs:
        doc_name = doc.get("document_name", "")
        chunk_idx = doc.get("chunk_index", -1)
        if chunk_idx < 0 or not doc_name:
            continue

        all_chunks = await asyncio.to_thread(vector_store.get_all_chunks)
        # 找到同文档的相邻 chunk
        same_doc = [
            c for c in all_chunks
            if c.get("document_name") == doc_name
        ]
        same_doc.sort(key=lambda c: c.get("chunk_index", 0))

        # 拉取前后 expand 个 chunk
        for c in same_doc:
            c_idx = c.get("chunk_index", -1)
            if abs(c_idx - chunk_idx) <= expand and c_idx != chunk_idx:
                cid = c.get("chunk_id", "")
                if cid not in seen_ids:
                    seen_ids.add(cid)
                    expanded.append({
                        "chunk_id": cid,
                        "content": c.get("content", ""),
                        "document_name": c.get("document_name", ""),
                        "page": c.get("page", 0),
                        "score": float(doc.get("score", 0)) * 0.8,  # 降权
                        "is_neighbor": True,
                    })

    return expanded
```

### 3.2.8 HyDE 查询增强

#### 最优解

让 LLM 根据 query 生成一段"假设答案"，用答案的向量去检索，而非直接用 query 的向量。原理：假设答案与真实文档在语义空间上更接近。

#### 实现方案

```python
# 新增: rag_orchestrator.py 中增加 HyDE 方法

async def _hyde_retrieve(self, query: str) -> list[dict]:
    """HyDE 检索增强：用假设答案的向量替代原始 query 向量。
    
    面试要点：能解释为什么 HyDE 有效——
    query 和 document 分别在"问题空间"和"文档空间"，
    语义鸿沟大。假设答案（hypothetical document）与真实文档
    处于同一语义空间，向量匹配更精准。
    """
    # 第 1 步：生成假设答案
    hyde_prompt = (
        f"请根据以下问题，生成一段假设性的答案（100-200字）。"
        f"不需要真实知识，只需要生成一段看起来像是文档中会出现的回答。"
        f"\n问题：{query}"
    )
    try:
        hyde_answer = await llm_service.chat(
            messages=[{"role": "user", "content": hyde_prompt}],
            system_prompt="你是知识库文档生成器。",
        )
    except Exception as exc:
        logger.warning("HyDE 生成失败，退回原始 query: %s", exc)
        hyde_answer = query

    # 第 2 步：用假设答案做向量检索
    return await asyncio.to_thread(
        retrieval_service.search, hyde_answer, settings.RETRIEVAL_TOP_K, False
    )
```

### 3.2.9 答案忠实度后验验证

#### 最优解

LLM 生成答案后，用另一个 LLM 调用（或同一 LLM 分步）逐句验证答案是否可以追溯到原文。如果某句无法追溯，标记为"潜在幻觉"并降级或重试。

#### 实现方案

```python
# 新增文件: backend/app/services/faithfulness_checker.py

class FaithfulnessChecker:
    """答案忠实度后验验证器。
    
    流程：
    1. 将 LLM 生成的答案拆分为独立陈述（statements）
    2. 对每个 statement，检查是否能在原文中找到支撑
    3. 无法验证的 statement 标记为 "potential_hallucination"
    4. 如果幻觉比例过高，触发降级话术
    """
    
    SYSTEM_PROMPT = "你是 RAG 答案忠实度验证器，只输出 JSON 数组。"
    
    async def check(
        self, answer: str, source_docs: list[dict]
    ) -> dict:
        """验证答案是否忠实于原文。"""
        # 拆分答案中的独立陈述
        statements = self._split_into_statements(answer)
        if not statements:
            return {"faithful": True, "hallucinations": []}
        
        # 构建原文参考
        source_text = "\n\n".join(
            f"[{i+1}] {d.get('content', '')[:500]}"
            for i, d in enumerate(source_docs)
        )
        
        # 逐条验证
        prompt = (
            "请判断以下每条陈述是否能在参考文档中找到支撑。"
            "输出 JSON 数组: [{\"index\":0,\"faithful\":true,\"evidence\":\"原文片段\"}]"
            f"\n\n参考文档:\n{source_text[:3000]}"
            f"\n\n待验证陈述:\n" + 
            "\n".join(f"[{i}] {s}" for i, s in enumerate(statements))
        )
        
        try:
            raw = await llm_service.chat(
                messages=[{"role": "user", "content": prompt}],
                system_prompt=self.SYSTEM_PROMPT,
            )
            results = self._parse_results(raw)
        except Exception:
            return {"faithful": True, "hallucinations": []}
        
        hallucinations = [
            statements[r["index"]] 
            for r in results 
            if not r.get("faithful", True) and r["index"] < len(statements)
        ]
        
        return {
            "faithful": len(hallucinations) == 0,
            "hallucinations": hallucinations,
            "hallucination_ratio": len(hallucinations) / max(len(statements), 1),
        }
    
    def _split_into_statements(self, text: str) -> list[str]:
        import re
        return [s.strip() for s in re.split(r'[。！？]', text) if s.strip()]
    
    def _parse_results(self, text: str) -> list[dict]:
        import re, json
        match = re.search(r"\[[\s\S]*\]", text.strip())
        return json.loads(match.group(0)) if match else []
```
